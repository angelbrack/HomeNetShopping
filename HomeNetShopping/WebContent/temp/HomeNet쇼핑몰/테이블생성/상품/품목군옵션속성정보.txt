/* 품목군 옵션 속성정보 */
DROP TABLE TB_ATTR_GRP_CD 
	CASCADE CONSTRAINTS;

/* 품목군 옵션 속성정보 */
CREATE TABLE TB_ATTR_GRP_CD (
	OPT_CD VARCHAR2(100) NOT NULL, /* 옵션코드 */
	OPT_NM VARCHAR2(100) NOT NULL, /* 옵션명 */
	USE_YN VARCHAR2(1) NOT NULL, /* 사용여부 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM VARCHAR2(14) NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM VARCHAR2(14) NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_ATTR_GRP_CD IS '품목군 옵션 속성정보';

COMMENT ON COLUMN TB_ATTR_GRP_CD.OPT_CD IS '옵션코드';

COMMENT ON COLUMN TB_ATTR_GRP_CD.OPT_NM IS '옵션명';

COMMENT ON COLUMN TB_ATTR_GRP_CD.USE_YN IS '사용여부';

COMMENT ON COLUMN TB_ATTR_GRP_CD.WRT_PN_NO IS '등록자ID';

COMMENT ON COLUMN TB_ATTR_GRP_CD.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_ATTR_GRP_CD.WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_ATTR_GRP_CD.UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_ATTR_GRP_CD.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_ATTR_GRP_CD.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX ATTR_GRP_CD_PK
	ON TB_ATTR_GRP_CD (
		OPT_CD ASC
	);

ALTER TABLE TB_ATTR_GRP_CD
	ADD
		CONSTRAINT ATTR_GRP_CD_PK
		PRIMARY KEY (
			OPT_CD
		);
