/* 품목군 옵션정보 */
DROP TABLE TB_GOODS_ARTC_OPT_INFO  
	CASCADE CONSTRAINTS;

/* 품목군 옵션정보 */
CREATE TABLE TB_GOODS_ARTC_OPT_INFO  (
	OPT_CD VARCHAR2(100) NOT NULL, /* 옵션코드 */
	ARTC_CD VARCHAR2(100) NOT NULL, /* 품목코드 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM VARCHAR2(14) NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM VARCHAR2(14) NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_GOODS_ARTC_OPT_INFO  IS '품목군 옵션정보';

COMMENT ON COLUMN TB_GOODS_ARTC_OPT_INFO .OPT_CD IS '옵션코드';

COMMENT ON COLUMN TB_GOODS_ARTC_OPT_INFO .ARTC_CD IS '품목코드';

COMMENT ON COLUMN TB_GOODS_ARTC_OPT_INFO .WRT_PN_NO IS '등록자ID';

COMMENT ON COLUMN TB_GOODS_ARTC_OPT_INFO .WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_GOODS_ARTC_OPT_INFO .WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_GOODS_ARTC_OPT_INFO .UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_GOODS_ARTC_OPT_INFO .UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_GOODS_ARTC_OPT_INFO .UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX GOODS_ARTC_OPT_INFO_PK
	ON TB_GOODS_ARTC_OPT_INFO  (
		OPT_CD ASC,
		ARTC_CD ASC
	);

CREATE INDEX GOODS_ARTC_OPT_INFO_FK1
	ON TB_GOODS_ARTC_OPT_INFO  (
		ARTC_CD ASC
	);

CREATE INDEX GOODS_ARTC_OPT_INFO_FK2
	ON TB_GOODS_ARTC_OPT_INFO  (
		OPT_CD ASC
	);

ALTER TABLE TB_GOODS_ARTC_OPT_INFO 
	ADD
		CONSTRAINT GOODS_ARTC_OPT_INFO_PK
		PRIMARY KEY (
			OPT_CD,
			ARTC_CD
		);

ALTER TABLE TB_GOODS_ARTC_OPT_INFO 
	ADD
		CONSTRAINT GOODS_ARTC_OPT_INFO_FK1
		FOREIGN KEY (
			ARTC_CD
		)
		REFERENCES TB_GOODS_ARTC_CD (
			ARTC_CD
		);

ALTER TABLE TB_GOODS_ARTC_OPT_INFO 
	ADD
		CONSTRAINT GOODS_ARTC_OPT_INFO_FK2
		FOREIGN KEY (
			OPT_CD
		)
		REFERENCES TB_ATTR_GRP_CD (

		);
