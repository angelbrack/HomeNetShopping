/* 공통첨부 */
CREATE TABLE TB_ATTACH_BASE (
    APND_FILE_NO NUMBER(10) NOT NULL, /* 첨부파일번호 */
    WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
    WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
    WRT_DTTM VARCHAR2(14) NOT NULL, /* 작성일시 */
    UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
    UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
    UPDT_DTTM VARCHAR2(14) NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_ATTACH_BASE IS '공통첨부';

COMMENT ON COLUMN TB_ATTACH_BASE.APND_FILE_NO IS '첨부파일번호';

COMMENT ON COLUMN TB_ATTACH_BASE.WRT_PN_NO IS '등록자ID';

COMMENT ON COLUMN TB_ATTACH_BASE.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_ATTACH_BASE.WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_ATTACH_BASE.UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_ATTACH_BASE.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_ATTACH_BASE.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX TB_ATTACH_BASE_PK
    ON TB_ATTACH_BASE (
        APND_FILE_NO ASC
    );

ALTER TABLE TB_ATTACH_BASE
    ADD
        CONSTRAINT TB_ATTACH_BASE_PK
        PRIMARY KEY (
            APND_FILE_NO
        );
