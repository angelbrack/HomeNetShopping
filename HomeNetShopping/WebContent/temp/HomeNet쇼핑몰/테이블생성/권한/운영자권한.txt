/* 운영자권한 */
DROP TABLE TB_OPTR_AUTH 
	CASCADE CONSTRAINTS;

/* 운영자권한 */
CREATE TABLE TB_OPTR_AUTH (
	OPTR_AUTH_NO NUMBER(10) NOT NULL, /* 운영자권한번호 */
	OPTR_AUTH_NM VARCHAR2(200) NOT NULL, /* 운영자권한명 */
	USER_AUTH_CD VARCHAR2(10), /* 사용자권한코드 */
	USE_YN VARCHAR2(1) NOT NULL, /* 사용여부 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM DATE NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM DATE NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_OPTR_AUTH IS '운영자권한';

COMMENT ON COLUMN TB_OPTR_AUTH.OPTR_AUTH_NO IS '운영자권한번호';

COMMENT ON COLUMN TB_OPTR_AUTH.OPTR_AUTH_NM IS '운영자권한명';

COMMENT ON COLUMN TB_OPTR_AUTH.USER_AUTH_CD IS '사용자권한코드';

COMMENT ON COLUMN TB_OPTR_AUTH.USE_YN IS '사용여부';

COMMENT ON COLUMN TB_OPTR_AUTH.WRT_PN_NO IS '작성자번호';

COMMENT ON COLUMN TB_OPTR_AUTH.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_OPTR_AUTH.WRT_DTTM IS '작성일시';

COMMENT ON COLUMN TB_OPTR_AUTH.UPDT_PN_NO IS '수정자번호';

COMMENT ON COLUMN TB_OPTR_AUTH.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_OPTR_AUTH.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX TB_OPTR_AUTH_PK
	ON TB_OPTR_AUTH (
		OPTR_AUTH_NO ASC
	);

ALTER TABLE TB_OPTR_AUTH
	ADD
		CONSTRAINT TB_OPTR_AUTH_PK
		PRIMARY KEY (
			OPTR_AUTH_NO
		);
