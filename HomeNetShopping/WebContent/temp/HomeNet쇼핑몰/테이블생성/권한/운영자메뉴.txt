/* 운영자메뉴 */
DROP TABLE TB_OPTR_MENU_INFO 
	CASCADE CONSTRAINTS;

/* 운영자메뉴 */
CREATE TABLE TB_OPTR_MENU_INFO (
	MENU_NO NUMBER(10) NOT NULL, /* 메뉴번호 */
	HGRK_MENU_NO NUMBER(10) NOT NULL, /* 상위메뉴번호 */
	OPTR_PGM_NO NUMBER(10), /* 운영자프로그램번호 */
	MENU_NM VARCHAR2(200) NOT NULL, /* 메뉴명 */
	SORT_OR NUMBER(3) NOT NULL, /* 정렬순서 */
	USE_YN VARCHAR2(1) NOT NULL, /* 사용여부 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM DATE NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM DATE NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_OPTR_MENU_INFO IS '운영자메뉴';

COMMENT ON COLUMN TB_OPTR_MENU_INFO.MENU_NO IS '메뉴번호';

COMMENT ON COLUMN TB_OPTR_MENU_INFO.HGRK_MENU_NO IS '상위메뉴번호';

COMMENT ON COLUMN TB_OPTR_MENU_INFO.OPTR_PGM_NO IS '운영자프로그램번호';

COMMENT ON COLUMN TB_OPTR_MENU_INFO.MENU_NM IS '메뉴명';

COMMENT ON COLUMN TB_OPTR_MENU_INFO.SORT_OR IS '정렬순서';

COMMENT ON COLUMN TB_OPTR_MENU_INFO.USE_YN IS '사용여부';

COMMENT ON COLUMN TB_OPTR_MENU_INFO.WRT_PN_NO IS '작성자번호';

COMMENT ON COLUMN TB_OPTR_MENU_INFO.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_OPTR_MENU_INFO.WRT_DTTM IS '작성일시';

COMMENT ON COLUMN TB_OPTR_MENU_INFO.UPDT_PN_NO IS '수정자번호';

COMMENT ON COLUMN TB_OPTR_MENU_INFO.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_OPTR_MENU_INFO.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX TB_OPTR_MENU_INFO_PK
	ON TB_OPTR_MENU_INFO (
		MENU_NO ASC
	);

CREATE INDEX TB_OPTR_MENU_INFO_FK1
	ON TB_OPTR_MENU_INFO (
		OPTR_PGM_NO ASC
	);

CREATE INDEX TB_OPTR_MENU_INFO_IX1
	ON TB_OPTR_MENU_INFO (
		MENU_NO ASC,
		HGRK_MENU_NO ASC
	);

ALTER TABLE TB_OPTR_MENU_INFO
	ADD
		CONSTRAINT TB_OPTR_MENU_INFO_PK
		PRIMARY KEY (
			MENU_NO
		);

ALTER TABLE TB_OPTR_MENU_INFO
	ADD
		CONSTRAINT TB_OPTR_MENU_INFO_FK1
		FOREIGN KEY (
			OPTR_PGM_NO
		)
		REFERENCES TB_OPTR_PGM_INFO (
			OPTR_PGM_NO
		);
