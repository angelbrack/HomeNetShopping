DROP TABLE TB_OPTR_PGM_INFO 
	CASCADE CONSTRAINTS;

/* 운영자프로그램 */
CREATE TABLE TB_OPTR_PGM_INFO (
	OPTR_PGM_NO NUMBER(10) NOT NULL, /* 운영자프로그램번호 */
	OPTR_PGM_ID VARCHAR2(100) NOT NULL, /* 운영자프로그램ID */
	OPTR_PGM_URL_V VARCHAR2(255), /* 운영자프로그램URL값 */
	OPTR_PGM_NM VARCHAR2(200) NOT NULL, /* 운영자프로그램명 */
	OPTR_PGM_GRP_NO NUMBER(10), /* 운영자프로그램그룹번호 */
	MENU_USE_YN VARCHAR2(1), /* 메뉴사용여부 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM VARCHAR2(14) NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM VARCHAR2(14) NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_OPTR_PGM_INFO IS '운영자프로그램';

COMMENT ON COLUMN TB_OPTR_PGM_INFO.OPTR_PGM_NO IS '운영자프로그램번호';

COMMENT ON COLUMN TB_OPTR_PGM_INFO.OPTR_PGM_ID IS '운영자프로그램ID';

COMMENT ON COLUMN TB_OPTR_PGM_INFO.OPTR_PGM_URL_V IS '운영자프로그램URL';

COMMENT ON COLUMN TB_OPTR_PGM_INFO.OPTR_PGM_NM IS '운영자프로그램명';

COMMENT ON COLUMN TB_OPTR_PGM_INFO.OPTR_PGM_GRP_NO IS '운영자프로그램그룹번호';

COMMENT ON COLUMN TB_OPTR_PGM_INFO.MENU_USE_YN IS '메뉴사용여부';

COMMENT ON COLUMN TB_OPTR_PGM_INFO.WRT_PN_NO IS '작성자번호';

COMMENT ON COLUMN TB_OPTR_PGM_INFO.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_OPTR_PGM_INFO.WRT_DTTM IS '작성일시';

COMMENT ON COLUMN TB_OPTR_PGM_INFO.UPDT_PN_NO IS '수정자번호';

COMMENT ON COLUMN TB_OPTR_PGM_INFO.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_OPTR_PGM_INFO.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX TB_OPTR_PGM_INFO_PK
	ON TB_OPTR_PGM_INFO (
		OPTR_PGM_NO ASC
	);

CREATE INDEX TB_OPTR_PGM_INFO_IX1
	ON TB_OPTR_PGM_INFO (
		OPTR_PGM_ID ASC
	);
