DROP TABLE TB_OPTR_PGM_AUTH

/* 운영자프로그램권한 */
CREATE TABLE TB_OPTR_PGM_AUTH (
	OPTR_AUTH_NO NUMBER(10) NOT NULL, /* 운영자권한번호 */
	OPTR_PGM_NO NUMBER(10) NOT NULL, /* 운영자프로그램번호 */

	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM VARCHAR2(14) NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM VARCHAR2(14) NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_OPTR_PGM_AUTH IS '운영자프로그램권한';

COMMENT ON COLUMN TB_OPTR_PGM_AUTH.OPTR_AUTH_NO IS '운영자권한번호';

COMMENT ON COLUMN TB_OPTR_PGM_AUTH.OPTR_PGM_NO IS '운영자프로그램번호';

COMMENT ON COLUMN TB_OPTR_PGM_AUTH.WRT_PN_NO IS '작성자번호';

COMMENT ON COLUMN TB_OPTR_PGM_AUTH.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_OPTR_PGM_AUTH.WRT_DTTM IS '작성일시';

COMMENT ON COLUMN TB_OPTR_PGM_AUTH.UPDT_PN_NO IS '수정자번호';

COMMENT ON COLUMN TB_OPTR_PGM_AUTH.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_OPTR_PGM_AUTH.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX TB_OPTR_PGM_AUTH_PK
	ON TB_OPTR_PGM_AUTH (

	);

	);
