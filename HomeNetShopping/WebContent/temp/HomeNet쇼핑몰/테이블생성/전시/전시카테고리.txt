/* 전시카테고리 */
DROP TABLE TB_DISP_GOODS_INFO 
	CASCADE CONSTRAINTS;

/* 전시카테고리 */
CREATE TABLE TB_DISP_GOODS_INFO (
	GOODS_NO NUMBER(10) NOT NULL, /* 상품번호 */
	DISP_NO NUMBER(10) NOT NULL, /* 전시번호 */
	DISP_PRIO_RNK NUMBER(10) NOT NULL, /* 전시우선순위 */
	USE_YN VARCHAR2(1) NOT NULL, /* 사용여부 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM DATE NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM VARCHAR2(14) NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_DISP_GOODS_INFO IS '전시카테고리';

COMMENT ON COLUMN TB_DISP_GOODS_INFO.GOODS_NO IS '상품번호';

COMMENT ON COLUMN TB_DISP_GOODS_INFO.DISP_NO IS '전시번호';

COMMENT ON COLUMN TB_DISP_GOODS_INFO.DISP_PRIO_RNK IS '전시우선순위';

COMMENT ON COLUMN TB_DISP_GOODS_INFO.USE_YN IS '사용여부';

COMMENT ON COLUMN TB_DISP_GOODS_INFO.WRT_PN_NO IS '등록자ID';

COMMENT ON COLUMN TB_DISP_GOODS_INFO.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_DISP_GOODS_INFO.WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_DISP_GOODS_INFO.UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_DISP_GOODS_INFO.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_DISP_GOODS_INFO.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX TB_DISP_GOODS_INFO_PK
	ON TB_DISP_GOODS_INFO (
		GOODS_NO ASC,
		DISP_NO ASC
	);

CREATE INDEX TB_DISP_GOODS_INFO_FK1
	ON TB_DISP_GOODS_INFO (
		GOODS_NO ASC
	);

CREATE INDEX TB_DISP_GOODS_INFO_FK2
	ON TB_DISP_GOODS_INFO (
		DISP_NO ASC
	);

ALTER TABLE TB_DISP_GOODS_INFO
	ADD
		CONSTRAINT TB_DISP_GOODS_INFO_PK
		PRIMARY KEY (
			GOODS_NO,
			DISP_NO
		);

ALTER TABLE TB_DISP_GOODS_INFO
	ADD
		CONSTRAINT TB_DISP_GOODS_INFO_FK1
		FOREIGN KEY (
			GOODS_NO
		)
		REFERENCES TB_GOODS_BASE (
			GOODS_NO
		);

ALTER TABLE TB_DISP_GOODS_INFO
	ADD
		CONSTRAINT TB_DISP_GOODS_INFO_FK2
		FOREIGN KEY (
			DISP_NO
		)
		REFERENCES TB_DISP_SHOP_BASE1 (
			DISP_NO
		);
