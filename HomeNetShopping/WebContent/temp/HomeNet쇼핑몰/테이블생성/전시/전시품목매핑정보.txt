/* 전시품목코드매핑정보 */
DROP TABLE TB_CAT_ARTC_REL_INFO 
	CASCADE CONSTRAINTS;

/* 전시품목코드매핑정보 */
CREATE TABLE TB_CAT_ARTC_REL_INFO (
	ARTC_CD VARCHAR2(10) NOT NULL, /* 품목코드 */
	DISP_NO NUMBER(10) NOT NULL, /* 전시번호 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM VARCHAR2(14) NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM VARCHAR2(14) NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_CAT_ARTC_REL_INFO IS '전시품목코드매핑정보';

COMMENT ON COLUMN TB_CAT_ARTC_REL_INFO.ARTC_CD IS '품목코드';

COMMENT ON COLUMN TB_CAT_ARTC_REL_INFO.DISP_NO IS '전시번호';

COMMENT ON COLUMN TB_CAT_ARTC_REL_INFO.WRT_PN_NO IS '등록자ID';

COMMENT ON COLUMN TB_CAT_ARTC_REL_INFO.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_CAT_ARTC_REL_INFO.WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_CAT_ARTC_REL_INFO.UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_CAT_ARTC_REL_INFO.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_CAT_ARTC_REL_INFO.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX CAT_ARTC_REL_INFO_PK
	ON TB_CAT_ARTC_REL_INFO (
		ARTC_CD ASC,
		DISP_NO ASC
	);

CREATE INDEX CAT_ARTC_REL_INFO_FK1
	ON TB_CAT_ARTC_REL_INFO (
		ARTC_CD ASC
	);

CREATE INDEX CAT_ARTC_REL_INFO_FK2
	ON TB_CAT_ARTC_REL_INFO (
		DISP_NO ASC
	);

ALTER TABLE TB_CAT_ARTC_REL_INFO
	ADD
		CONSTRAINT CAT_ARTC_REL_INFO_PK
		PRIMARY KEY (
			ARTC_CD,
			DISP_NO
		);

ALTER TABLE TB_CAT_ARTC_REL_INFO
	ADD
		CONSTRAINT CAT_ARTC_REL_INFO_FK1
		FOREIGN KEY (
			ARTC_CD
		)
		REFERENCES TB_GOODS_ARTC_CD (
			ARTC_CD
		);

ALTER TABLE TB_CAT_ARTC_REL_INFO
	ADD
		CONSTRAINT CAT_ARTC_REL_INFO_FK2
		FOREIGN KEY (
			DISP_NO
		)
		REFERENCES TB_DISP_SHOP_BASE (
			DISP_NO
		);
