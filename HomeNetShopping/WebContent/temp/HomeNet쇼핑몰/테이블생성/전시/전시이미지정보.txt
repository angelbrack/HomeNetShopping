/* 전시이미지정보 */
DROP TABLE TB_DISP_IMG_INFO 
	CASCADE CONSTRAINTS;

/* 전시이미지정보 */
CREATE TABLE TB_DISP_IMG_INFO (
	DISP_NO NUMBER(10) NOT NULL, /* 전시번호 */
	DISP_SHOP_SCT_CD VARCHAR2(10) NOT NULL, /* 전시매장구분코드 */
	IMG_SN NUMBER(10) NOT NULL, /* 이미지일련번호 */
	DISP_IMG_TP_CD VARCHAR2(10) NOT NULL, /* 전시이미지유형코드 */
	BNR_IMG_PATH_NM VARCHAR2(200) NOT NULL, /* 배너이미지경로명 */
	BNR_IMG_FILE_NM VARCHAR2(200) NOT NULL, /* 배너이미지파일명 */
	LNK_URL_ADDR VARCHAR2(200), /* 연결 URL주소 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM VARCHAR2(14) NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM VARCHAR2(14) NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_DISP_IMG_INFO IS '전시이미지정보';

COMMENT ON COLUMN TB_DISP_IMG_INFO.DISP_NO IS '전시번호';

COMMENT ON COLUMN TB_DISP_IMG_INFO.DISP_SHOP_SCT_CD IS '전시매장구분코드';

COMMENT ON COLUMN TB_DISP_IMG_INFO.IMG_SN IS '이미지일련번호';

COMMENT ON COLUMN TB_DISP_IMG_INFO.DISP_IMG_TP_CD IS '전시이미지유형코드';

COMMENT ON COLUMN TB_DISP_IMG_INFO.BNR_IMG_PATH_NM IS '배너이미지경로명';

COMMENT ON COLUMN TB_DISP_IMG_INFO.BNR_IMG_FILE_NM IS '배너이미지파일명';

COMMENT ON COLUMN TB_DISP_IMG_INFO.LNK_URL_ADDR IS '연결 URL주소';

COMMENT ON COLUMN TB_DISP_IMG_INFO.WRT_PN_NO IS '등록자ID';

COMMENT ON COLUMN TB_DISP_IMG_INFO.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_DISP_IMG_INFO.WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_DISP_IMG_INFO.UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_DISP_IMG_INFO.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_DISP_IMG_INFO.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX DISP_IMG_INFO_PK
	ON TB_DISP_IMG_INFO (
		DISP_NO ASC,
		DISP_SHOP_SCT_CD ASC,
		IMG_SN ASC
	);

CREATE INDEX DISP_IMG_INFO_FK1
	ON TB_DISP_IMG_INFO (
		DISP_NO ASC
	);

ALTER TABLE TB_DISP_IMG_INFO
	ADD
		CONSTRAINT DISP_IMG_INFO_PK
		PRIMARY KEY (
			DISP_NO,
			DISP_SHOP_SCT_CD,
			IMG_SN
		);

ALTER TABLE TB_DISP_IMG_INFO
	ADD
		CONSTRAINT DISP_IMG_INFO_FK1
		FOREIGN KEY (
			DISP_NO
		)
		REFERENCES TB_DISP_SHOP_BASE (
			DISP_NO
		);
