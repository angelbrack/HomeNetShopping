/* 전시기본정보 */
DROP TABLE TB_DISP_SHOP_BASE 
	CASCADE CONSTRAINTS;

/* 전시기본정보 */
CREATE TABLE TB_DISP_SHOP_BASE (
	DISP_NO NUMBER(10) NOT NULL, /* 전시번호 */
	UPR_DISP_NO NUMBER(10), /* 상위전시번호 */
	DPML_NO VARCHAR2(10), /* 전시몰번호 */
	DISP_NM VARCHAR2(200) NOT NULL, /* 전시명 */
	DISP_GNB_NM VARCHAR2(200), /* 전시 GNB명 */
	GNB_EXP_METH_CD VARCHAR2(10), /* 전시 GNB노출방식 */
	DISP_TIT_NM VARCHAR2(200), /* 전시 제목명 */
	DISP_TIT_EXP_METH_CD VARCHAR2(10), /* 전시제목노출방식 */
	SHOP_TP_CD VARCHAR2(10), /* 매장유형코드 */
	SHOP_DESC_CONT VARCHAR2(4000), /* 매장설명내용 */
	DPTH_NO NUMBER(10), /* 깊이번호 */
	DISP_PRIO_RNK NUMBER(10), /* 전시우선순위 */
	USE_YN VARCHAR2(1), /* 사용여부 */
	DISP_YN VARCHAR2(1), /* 전시여부 */
	TLWT_LF_YN VARCHAR2(1), /* 최하위리프여부 */
	PRT_TP_CD VARCHAR2(10), /* 출력유형코드 */
	LST_SORT_CD VARCHAR2(10), /* 리스트정렬코드 */
	MOV_FRME_CD VARCHAR2(10), /* 이동프레임코드 */
	LNK_URL_ADDR VARCHAR2(200), /* 연결 url주소 */
	LNK_SPDP_HH_NO NUMBER(10), /* 연결 기획전 전시번호  */
	DISP_LRG_NO NUMBER(10), /* 전시대번호 */
	DISP_MID_NO NUMBER(10), /* 전시중번호 */
	DISP_SML_NO NUMBER(10), /* 전시소번호 */
	DISP_THN_NO NUMBER(10), /* 전시세번호 */
	MENU_USE_YN VARCHAR2(1), /* 메뉴사용여부 */
	TMPL_NO NUMBER(10), /* 템플릿번호 */
	PPP_SN NUMBER(10), /* 팝업일련번호 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM VARCHAR2(14) NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM VARCHAR2(14) NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_DISP_SHOP_BASE IS '전시기본정보';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.DISP_NO IS '전시번호';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.UPR_DISP_NO IS '상위전시번호';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.DPML_NO IS '전시몰번호';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.DISP_NM IS '전시명';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.DISP_GNB_NM IS '전시 GNB명';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.GNB_EXP_METH_CD IS '전시 GNB노출방식';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.DISP_TIT_NM IS '전시 제목명';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.DISP_TIT_EXP_METH_CD IS '전시제목노출방식';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.SHOP_TP_CD IS '매장유형코드';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.SHOP_DESC_CONT IS '매장설명내용';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.DPTH_NO IS '깊이번호';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.DISP_PRIO_RNK IS '전시우선순위';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.USE_YN IS '사용여부';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.DISP_YN IS '전시여부';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.TLWT_LF_YN IS '최하위리프여부';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.PRT_TP_CD IS '출력유형코드';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.LST_SORT_CD IS '리스트정렬코드';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.MOV_FRME_CD IS '이동프레임코드';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.LNK_URL_ADDR IS '연결 url주소';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.LNK_SPDP_HH_NO IS '연결 기획전 전시번호 ';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.DISP_LRG_NO IS '전시대번호';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.DISP_MID_NO IS '전시중번호';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.DISP_SML_NO IS '전시소번호';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.DISP_THN_NO IS '전시세번호';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.MENU_USE_YN IS '메뉴사용여부';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.TMPL_NO IS '템플릿번호';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_DISP_SHOP_BASE.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX TB_DISP_SHOP_BASE_PK
	ON TB_DISP_SHOP_BASE (
		DISP_NO ASC
	);

CREATE INDEX TB_DISP_SHOP_BASE_FK1
	ON TB_DISP_SHOP_BASE (
		DPML_NO ASC
	);

ALTER TABLE TB_DISP_SHOP_BASE
	ADD
		CONSTRAINT TB_DISP_SHOP_BASE_PK
		PRIMARY KEY (
			DISP_NO
		);

ALTER TABLE TB_DISP_SHOP_BASE
	ADD
		CONSTRAINT TB_DISP_SHOP_BASE_FK1
		FOREIGN KEY (
			DPML_NO
		)
		REFERENCES TB_DPML_BASE (

		);
