/* 전시몰정보 */
DROP TABLE TB_DPML_BASE 
	CASCADE CONSTRAINTS;

/* 전시몰정보 */
CREATE TABLE TB_DPML_BASE (
	DPML_NO VARCHAR2(10) NOT NULL, /* 전시몰번호 */
	DPML_NM VARCHAR2(50) NOT NULL, /* 전시몰명 */
	MALL_TP_CD VARCHAR2(10) NOT NULL, /* 몰유형코드 */
	PRT_TP_CD VARCHAR2(10) NOT NULL, /* 출력유형코드 */
	MOV_FRME_CD VARCHAR2(10), /* 이동프레임코드 */
	DISP_YN VARCHAR2(1), /* 전시여부 */
	USE_YN VARCHAR2(1), /* 사용여부 */
	LNK_URL_ADDR VARCHAR2(200), /* 연결 url주소 */
	LNK_SPDP_HH_NO VARCHAR2(200), /* 연결기획전시번호 */
	TMPL_NO NUMBER(10), /* 템플릿번호 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM VARCHAR2(14) NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM VARCHAR2(14) NOT NULL, /* 수정일시 */
	GOODS_NO NUMBER(10) /* 상품번호 */
);

COMMENT ON TABLE TB_DPML_BASE IS '전시몰정보';

COMMENT ON COLUMN TB_DPML_BASE.DPML_NO IS '전시몰번호';

COMMENT ON COLUMN TB_DPML_BASE.DPML_NM IS '전시몰명';

COMMENT ON COLUMN TB_DPML_BASE.MALL_TP_CD IS '몰유형코드';

COMMENT ON COLUMN TB_DPML_BASE.PRT_TP_CD IS '출력유형코드';

COMMENT ON COLUMN TB_DPML_BASE.MOV_FRME_CD IS '이동프레임코드';

COMMENT ON COLUMN TB_DPML_BASE.DISP_YN IS '전시여부';

COMMENT ON COLUMN TB_DPML_BASE.USE_YN IS '사용여부';

COMMENT ON COLUMN TB_DPML_BASE.LNK_URL_ADDR IS '연결 url주소';

COMMENT ON COLUMN TB_DPML_BASE.LNK_SPDP_HH_NO IS '연결기획전시번호';

COMMENT ON COLUMN TB_DPML_BASE.TMPL_NO IS '템플릿번호';

COMMENT ON COLUMN TB_DPML_BASE.WRT_PN_NO IS '등록자ID';

COMMENT ON COLUMN TB_DPML_BASE.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_DPML_BASE.WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_DPML_BASE.UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_DPML_BASE.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_DPML_BASE.UPDT_DTTM IS '수정일시';

COMMENT ON COLUMN TB_DPML_BASE.GOODS_NO IS '상품번호';

CREATE UNIQUE INDEX DPML_BASE_PK
	ON TB_DPML_BASE (
		DPML_NO ASC
	);

CREATE INDEX DPML_BASE_FK1
	ON TB_DPML_BASE (
		TMPL_NO ASC
	);

ALTER TABLE TB_DPML_BASE
	ADD
		CONSTRAINT DPML_BASE_PK
		PRIMARY KEY (
			DPML_NO
		);

ALTER TABLE TB_DPML_BASE
	ADD
		CONSTRAINT DPML_BASE_FK1
		FOREIGN KEY (
			TMPL_NO
		)
		REFERENCES TB_TMPL_BASE (
			TMPL_NO
		);
