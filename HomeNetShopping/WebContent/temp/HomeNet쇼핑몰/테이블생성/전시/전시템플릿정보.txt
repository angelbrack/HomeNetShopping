/* 전시템플릿정보 */
DROP TABLE TB_TMPL_BASE 
	CASCADE CONSTRAINTS;

/* 전시템플릿정보 */
CREATE TABLE TB_TMPL_BASE (
	TMPL_NO NUMBER(10) NOT NULL, /* 템플릿번호 */
	DPML_NO VARCHAR2(10) NOT NULL, /* 전시몰번호 */
	TMPL_TP_CD VARCHAR2(10) NOT NULL, /* 템플릿유형코드 */
	TMPL_NM VARCHAR2(200) NOT NULL, /* 템플릿명 */
	TMPL_PATH_NM VARCHAR2(200), /* 템플릿경로명 */
	TMPL_FILE_NM VARCHAR2(200), /* 템플릿파일명 */
	USE_YN VARCHAR2(1) NOT NULL, /* 사용여부 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM VARCHAR2(14) NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM VARCHAR2(14) NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_TMPL_BASE IS '전시템플릿정보';

COMMENT ON COLUMN TB_TMPL_BASE.TMPL_NO IS '템플릿번호';

COMMENT ON COLUMN TB_TMPL_BASE.DPML_NO IS '전시몰번호';

COMMENT ON COLUMN TB_TMPL_BASE.TMPL_TP_CD IS '템플릿유형코드';

COMMENT ON COLUMN TB_TMPL_BASE.TMPL_NM IS '템플릿명';

COMMENT ON COLUMN TB_TMPL_BASE.TMPL_PATH_NM IS '템플릿경로명';

COMMENT ON COLUMN TB_TMPL_BASE.TMPL_FILE_NM IS '템플릿파일명';

COMMENT ON COLUMN TB_TMPL_BASE.USE_YN IS '사용여부';

COMMENT ON COLUMN TB_TMPL_BASE.WRT_PN_NO IS '등록자ID';

COMMENT ON COLUMN TB_TMPL_BASE.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_TMPL_BASE.WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_TMPL_BASE.UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_TMPL_BASE.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_TMPL_BASE.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX TMPL_BASE_PK
	ON TB_TMPL_BASE (
		TMPL_NO ASC
	);

CREATE INDEX TMPL_BASE_FK1
	ON TB_TMPL_BASE (
		DPML_NO ASC
	);

ALTER TABLE TB_TMPL_BASE
	ADD
		CONSTRAINT TMPL_BASE_PK
		PRIMARY KEY (
			TMPL_NO
		);

ALTER TABLE TB_TMPL_BASE
	ADD
		CONSTRAINT TMPL_BASE_FK1
		FOREIGN KEY (
			DPML_NO
		)
		REFERENCES TB_DPML_BASE (
			DPML_NO
		);
