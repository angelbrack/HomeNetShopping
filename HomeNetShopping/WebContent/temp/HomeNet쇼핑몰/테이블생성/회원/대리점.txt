/* 대리점 */
CREATE TABLE TB_AGENCY (
    AGEN_NO NUMBER(10) NOT NULL, /* 대리점번호 */
    DIRC_NO NUMBER(10) NOT NULL, /* 본부장번호 */
    AGEN_NM VARCHAR2(100) NOT NULL, /* 대리점명 */
    WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
    WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
    WRT_DTTM DATE NOT NULL, /* 작성일시 */
    UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
    UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
    UPDT_DTTM DATE NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_AGENCY IS '대리점';

COMMENT ON COLUMN TB_AGENCY.AGEN_NO IS '대리점번호';

COMMENT ON COLUMN TB_AGENCY.DIRC_NO IS '본부장번호';

COMMENT ON COLUMN TB_AGENCY.AGEN_NM IS '대리점명';

COMMENT ON COLUMN TB_AGENCY.WRT_PN_NO IS '등록자ID';

COMMENT ON COLUMN TB_AGENCY.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_AGENCY.WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_AGENCY.UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_AGENCY.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_AGENCY.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX TB_AGENCY_PK
    ON TB_AGENCY (
        AGEN_NO ASC
    );

ALTER TABLE TB_AGENCY
    ADD
        CONSTRAINT TB_AGENCY_PK
        PRIMARY KEY (
            AGEN_NO
        );

ALTER TABLE TB_AGENCY
    ADD
        CONSTRAINT TB_AGENCY_FK1
        FOREIGN KEY (
            DIRC_NO
        )
        REFERENCES TB_CHEF_DIRC (
            DIRC_NO
        );
