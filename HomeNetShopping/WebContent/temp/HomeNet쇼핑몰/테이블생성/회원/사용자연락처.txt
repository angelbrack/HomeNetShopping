/* 사용자연락처 */
CREATE TABLE TB_USER_CTPLC (
	USER_NO NUMBER(10) NOT NULL, /* 사용자번호 */

	USER_CTPLC VARCHAR2(255) /* 사용자연락처 */
);

COMMENT ON TABLE TB_USER_CTPLC IS '사용자연락처';

COMMENT ON COLUMN TB_USER_CTPLC.USER_NO IS '사용자번호';

COMMENT ON COLUMN TB_USER_CTPLC.USER_CTPLC_DV_CD IS '사용자연락처구분코드';

COMMENT ON COLUMN TB_USER_CTPLC.USER_CTPLC IS '사용자연락처';

CREATE UNIQUE INDEX TB_USER_CTPLC_PK
	ON TB_USER_CTPLC (
		USER_NO ASC,
		USER_CTPLC_DV_CD ASC
	);

CREATE UNIQUE INDEX TB_USER_CTPLC_FK1
	ON TB_USER_CTPLC (
		USER_NO ASC
	);
