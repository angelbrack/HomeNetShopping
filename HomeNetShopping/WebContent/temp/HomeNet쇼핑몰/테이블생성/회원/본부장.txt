/* 본부장 */
CREATE TABLE TB_CHEF_DIRC (
    DIRC_NO NUMBER(10) NOT NULL, /* 본부장번호 */
    DIRC_NM VARCHAR2(100) NOT NULL, /* 본부장이름 */
    DIRC_CD VARCHAR2(10) NOT NULL, /* 본부구분코드 */
    WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
    WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
    WRT_DTTM DATE NOT NULL, /* 작성일시 */
    UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
    UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
    UPDT_DTTM DATE NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_CHEF_DIRC IS '본부장';

COMMENT ON COLUMN TB_CHEF_DIRC.DIRC_NO IS '본부장번호';

COMMENT ON COLUMN TB_CHEF_DIRC.DIRC_NM IS '본부장이름';

COMMENT ON COLUMN TB_CHEF_DIRC.DIRC_CD IS '본부구분코드';

COMMENT ON COLUMN TB_CHEF_DIRC.WRT_PN_NO IS '등록자ID';

COMMENT ON COLUMN TB_CHEF_DIRC.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_CHEF_DIRC.WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_CHEF_DIRC.UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_CHEF_DIRC.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_CHEF_DIRC.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX TB_CHEF_DIRC_PK
    ON TB_CHEF_DIRC (
        DIRC_NO ASC
    );

ALTER TABLE TB_CHEF_DIRC
    ADD
        CONSTRAINT TB_CHEF_DIRC_PK
        PRIMARY KEY (
            DIRC_NO
        );
