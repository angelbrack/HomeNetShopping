/* 공급업체 */
DROP TABLE TB_SUP_CORP_BASE 
	CASCADE CONSTRAINTS;

/* 공급업체 */
CREATE TABLE TB_SUP_CORP_BASE (
	SUP_CORP_NO NUMBER(10) NOT NULL, /* 공급업체번호 */
	SUP_CORP_NM VARCHAR2(100) NOT NULL, /* 공급업체명 */
	MAST_PHON VARCHAR2(20), /* 대표전화번호 */
	FAX_NO VARCHAR2(20), /* 팩스번호 */
	AEMP_NM VARCHAR2(100), /* 담당자이름 */
	MAST_EMAIL_ADDR VARCHAR2(100), /* 대표이메일주소 */
	CORP_SCT_CD VARCHAR2(10), /* 업체구분코드 */
	CUR_ENTR_YN VARCHAR2(1), /* 현재거래처여부 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM DATE NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM DATE NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_SUP_CORP_BASE IS '공급업체';

COMMENT ON COLUMN TB_SUP_CORP_BASE.SUP_CORP_NO IS '공급업체번호';

COMMENT ON COLUMN TB_SUP_CORP_BASE.SUP_CORP_NM IS '공급업체명';

COMMENT ON COLUMN TB_SUP_CORP_BASE.MAST_PHON IS '대표전화번호';

COMMENT ON COLUMN TB_SUP_CORP_BASE.FAX_NO IS '팩스번호';

COMMENT ON COLUMN TB_SUP_CORP_BASE.AEMP_NM IS '담당자이름';

COMMENT ON COLUMN TB_SUP_CORP_BASE.MAST_EMAIL_ADDR IS '대표이메일주소';

COMMENT ON COLUMN TB_SUP_CORP_BASE.CORP_SCT_CD IS '업체구분코드';

COMMENT ON COLUMN TB_SUP_CORP_BASE.CUR_ENTR_YN IS '현재거래처여부';

COMMENT ON COLUMN TB_SUP_CORP_BASE.WRT_PN_NO IS '등록자ID';

COMMENT ON COLUMN TB_SUP_CORP_BASE.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_SUP_CORP_BASE.WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_SUP_CORP_BASE.UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_SUP_CORP_BASE.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_SUP_CORP_BASE.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX TB_SUP_CORP_BASE_PK
	ON TB_SUP_CORP_BASE (
		SUP_CORP_NO ASC
	);

ALTER TABLE TB_SUP_CORP_BASE
	ADD
		CONSTRAINT TB_SUP_CORP_BASE_PK
		PRIMARY KEY (
			SUP_CORP_NO
		);
