/* 브랜드정보 */
DROP TABLE TB_BRND_BASE 
	CASCADE CONSTRAINTS;

/* 브랜드정보 */
CREATE TABLE TB_BRND_BASE (
	BRND_NO NUMBER(10) NOT NULL, /* 브랜드번호 */
	BRND_NM VARCHAR2(200) NOT NULL, /* 브랜드명 */
	BRND_KOR_NM VARCHAR2(200), /* 브랜드한글명 */
	BRND_ENG_NM VARCHAR2(200), /* 브랜드영문명 */
	BRND_DESC_CONT VARCHAR2(4000), /* 브랜드상세설명 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM VARCHAR2(14) NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM VARCHAR2(14) NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_BRND_BASE IS '브랜드정보';

COMMENT ON COLUMN TB_BRND_BASE.BRND_NO IS '브랜드번호';

COMMENT ON COLUMN TB_BRND_BASE.BRND_NM IS '브랜드명';

COMMENT ON COLUMN TB_BRND_BASE.BRND_KOR_NM IS '브랜드한글명';

COMMENT ON COLUMN TB_BRND_BASE.BRND_ENG_NM IS '브랜드영문명';

COMMENT ON COLUMN TB_BRND_BASE.BRND_DESC_CONT IS '브랜드상세설명';

COMMENT ON COLUMN TB_BRND_BASE.WRT_PN_NO IS '등록자ID';

COMMENT ON COLUMN TB_BRND_BASE.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_BRND_BASE.WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_BRND_BASE.UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_BRND_BASE.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_BRND_BASE.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX TB_BRND_BASE_PK
	ON TB_BRND_BASE (
		BRND_NO ASC
	);

ALTER TABLE TB_BRND_BASE
	ADD
		CONSTRAINT TB_BRND_BASE_PK
		PRIMARY KEY (
			BRND_NO
		);
