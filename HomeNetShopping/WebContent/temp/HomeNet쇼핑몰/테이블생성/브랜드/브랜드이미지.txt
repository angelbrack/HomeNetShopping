/* 브랜드 이미지정보 */
DROP TABLE TB_BRND_IMG_INFO 
	CASCADE CONSTRAINTS;

/* 브랜드 이미지정보 */
CREATE TABLE TB_BRND_IMG_INFO (
	IMG_NO NUMBER(10) NOT NULL, /* 이미지번호 */
	BRND_NO NUMBER(10) NOT NULL, /* 브랜드번호 */
	IMG_NM VARCHAR2(200) NOT NULL, /* 이미지명 */
	IMG_PATH_NM VARCHAR2(200) NOT NULL, /* 이미지경로명 */
	IMG_FILE_NM VARCHAR2(200) NOT NULL, /* 파일명 */
	WRT_PN_NO NUMBER(10) NOT NULL, /* 작성자번호 */
	WRT_PN_IP VARCHAR2(20) NOT NULL, /* 작성자IP */
	WRT_DTTM VARCHAR2(14) NOT NULL, /* 작성일시 */
	UPDT_PN_NO NUMBER(10) NOT NULL, /* 수정자번호 */
	UPDT_PN_IP VARCHAR2(20) NOT NULL, /* 수정자IP */
	UPDT_DTTM VARCHAR2(14) NOT NULL /* 수정일시 */
);

COMMENT ON TABLE TB_BRND_IMG_INFO IS '브랜드 이미지정보';

COMMENT ON COLUMN TB_BRND_IMG_INFO.IMG_NO IS '이미지번호';

COMMENT ON COLUMN TB_BRND_IMG_INFO.BRND_NO IS '브랜드번호';

COMMENT ON COLUMN TB_BRND_IMG_INFO.IMG_NM IS '이미지명';

COMMENT ON COLUMN TB_BRND_IMG_INFO.IMG_PATH_NM IS '이미지경로명';

COMMENT ON COLUMN TB_BRND_IMG_INFO.IMG_FILE_NM IS '파일명';

COMMENT ON COLUMN TB_BRND_IMG_INFO.WRT_PN_NO IS '등록자ID';

COMMENT ON COLUMN TB_BRND_IMG_INFO.WRT_PN_IP IS '작성자IP';

COMMENT ON COLUMN TB_BRND_IMG_INFO.WRT_DTTM IS '등록일시';

COMMENT ON COLUMN TB_BRND_IMG_INFO.UPDT_PN_NO IS '수정자ID';

COMMENT ON COLUMN TB_BRND_IMG_INFO.UPDT_PN_IP IS '수정자IP';

COMMENT ON COLUMN TB_BRND_IMG_INFO.UPDT_DTTM IS '수정일시';

CREATE UNIQUE INDEX TB_BRND_IMG_INFO_PK
	ON TB_BRND_IMG_INFO (
		IMG_NO ASC
	);

CREATE INDEX TB_BRND_IMG_INFO_FK1
	ON TB_BRND_IMG_INFO (
		BRND_NO ASC
	);

ALTER TABLE TB_BRND_IMG_INFO
	ADD
		CONSTRAINT TB_BRND_IMG_INFO_PK
		PRIMARY KEY (
			IMG_NO
		);

ALTER TABLE TB_BRND_IMG_INFO
	ADD
		CONSTRAINT TB_BRND_IMG_INFO_FK1
		FOREIGN KEY (
			BRND_NO
		)
		REFERENCES TB_BRND_BASE (
			BRND_NO
		);
